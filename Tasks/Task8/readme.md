```Racket
#lang racket

(define (chse x y z)
  (cond
    [(null? x)'()] 
    [(null? y)'()]
    [(null? y)'()]
    [ (if(= (car x) 1)
         (cons(car y)(chse(cdr x)(cdr y)(cdr z)))
         (cons(car z)(chse(cdr x)(cdr y)(cdr z))))]
    )
  )

; Test Cases
(displayln "***************** CHSE Results *****************\n")
```
